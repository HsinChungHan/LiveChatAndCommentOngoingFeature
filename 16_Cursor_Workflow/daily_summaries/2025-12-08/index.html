    private static let cacheKey = "IntegrationTestTokenCache"
    private static let validDuration: TimeInterval = 8 * 60 * 60  // 8 hours

    struct CachedToken: Codable {
        let accessToken: String
        let userId: String
        let timestamp: TimeInterval
    }

    static func getCachedToken() -> CachedToken? {
        guard let data = UserDefaults.standard.data(forKey: cacheKey),
              let cached = try? JSONDecoder().decode(CachedToken.self, from: data) else {
            return nil
        }

        let now = Date().timeIntervalSince1970
        if now - cached.timestamp < validDuration {
            return cached
        }
        return nil
    }

    static func saveToken(accessToken: String, userId: String) {
        let cached = CachedToken(
            accessToken: accessToken,
            userId: userId,
            timestamp: Date().timeIntervalSince1970
        )
        if let data = try? JSONEncoder().encode(cached) {
            UserDefaults.standard.set(data, forKey: cacheKey)
        }
    }
}
